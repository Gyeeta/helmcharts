Please be patient while the chart is being deployed...

PostgreSQL can be accessed via port {{ include "postgresql.service.port" . }} on the following DNS name from within your cluster:

    {{ include "postgresdb.fullname" . }}.{{ .Release.Namespace }}.svc.cluster.local 

To get the password for "postgres" user run:

    export POSTGRES_PASSWORD=$(kubectl get secret --namespace {{ .Release.Namespace }} {{ include "postgresdb.fullname" . }} -o jsonpath="{.data.postgres-password}" | base64 -d)


To connect to your database run the following command:

    kubectl run {{ include "postgresdb.fullname" . }}-client --rm --tty -i --restart='Never' --namespace {{ .Release.Namespace }} --image {{ include "postgresdb.image" . }} --env="PGPASSWORD=$POSTGRES_PASSWORD" \
      --command -- /postgresdb/bin/psql --host {{ include "postgresdb.fullname" . }} -U postgres -d postgres -p {{ include "postgresdb_config.service.port" . }}

To connect to your database from outside the cluster execute the following commands:

{{- if contains "NodePort" .Values.postgresdb_config.service.type }}

    export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
    export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "postgresdb.fullname" . }})
    PGPASSWORD="$POSTGRES_PASSWORD" psql --host $NODE_IP --port $NODE_PORT -U postgres -d postgres

{{- else if contains "LoadBalancer" .Values.postgresdb_config.service.type }}

  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        Watch the status with: 'kubectl get svc --namespace {{ .Release.Namespace }} -w {{ include "postgresdb.fullname" . }}'

    export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "postgresdb.fullname" . }} --template "{{ "{{ range (index .status.loadBalancer.ingress 0) }}{{ . }}{{ end }}" }}")
    PGPASSWORD="$POSTGRES_PASSWORD" psql --host $SERVICE_IP --port {{ include "postgresdb_config.service.port" . }} -U postgres -d postgres

{{- else if contains "ClusterIP" .Values.postgresdb_config.service.type }}

    kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "postgresdb.fullname" . }} {{ include "postgresdb_config.service.port" . }}:{{ include "postgresdb_config.service.port" . }} &
    PGPASSWORD="$POSTGRES_PASSWORD" psql --host 127.0.0.1 -U postgres -d postgres -p {{ include "postgresdb_config.service.port" . }}

{{- end }}
